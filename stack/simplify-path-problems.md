var simplifyPath = function (path) {
  const strs = path.split('/');
  const stack = [];

  for (const str of strs) {
    if (str === '' || str === '.') continue;
    if (str === '..') {
      if (stack.length > 0) stack.pop();
    } else {
      stack.push(str);
    }
  }

  return '/' + stack.join('/');
};
